def deposit(self, timestamp, account_id, amount):    #SK
    self._process_cashbacks(timestamp)               #SK  # L3 加
    if account_id not in self.accounts:              #SK  # L1
        return None                                  #SK  # L1
    acc = self.accounts[account_id]                  #SK  # L1 攞短名
    acc["balance"] += amount                         #CH  # L1 做嘢
    acc["history"].append((timestamp, acc["balance"])) #CH # L4 加
    return acc["balance"]                            #SK  # L1

transfer / 雙 entity 操作（完整 example）

def transfer(self, timestamp, source_id, target_id, amount): #SK
    self._process_cashbacks(timestamp)               #SK  # L3 加
    if source_id not in self.accounts or target_id not in self.accounts: #SK # L1
        return None                                  #SK  # L1
    if source_id == target_id:                       #SK  # L1
        return None                                  #SK  # L1
    s_acc = self.accounts[source_id]                 #SK  # L1
    t_acc = self.accounts[target_id]                 #SK  # L1
    if s_acc["balance"] < amount:                    #SK  # L1
        return None                                  #SK  # L1
    s_acc["balance"] -= amount                       #CH  # L1
    s_acc["outgoing"] += amount                      #CH  # L2 加
    s_acc["history"].append((timestamp, s_acc["balance"])) #CH # L4 加
    t_acc["balance"] += amount                       #CH  # L1
    t_acc["history"].append((timestamp, t_acc["balance"])) #CH # L4 加
    return s_acc["balance"]                          #SK  # L1
Dict vs Dot： d["balance"] 唔係 d.balance
Variable vs String： d[account_id] 用 variable，d["account_id"] 搵 literal string
Dict 係 pointer： acc = self.accounts[id] 改 acc 會改原本

排序 + format（完整 example）

def top_spenders(self, timestamp, n):                #SK
    self._process_cashbacks(timestamp)               #SK
    sorted_items = dict(sorted(                      #SK
        self.accounts.items(),                       #SK
        key=lambda x: (-x[1]["outgoing"], x[0])      #SK  # desc by value, asc by name
    ))
    result = []                                      #SK
    for account_id, acc in sorted_items.items():     #SK
        result.append(f"{account_id}({acc['outgoing']})") #CH
    return result[:n]                                #SK  # [:n] 自動處理 n 太大
sorted() 人話：「將呢堆嘢，按呢個準則排」。負號 = 反轉。
x[0] = key（"alice"），x[1] = value（成個 dict）
f-string： f"{name}({value})" → "alice(500)"

def _process_cashbacks(self, timestamp):             #SK
    for pid, p in self.payment.items():              #SK
        if not p["received"] and timestamp >= p["cashback_time"]: #SK
            self.accounts[p["account_id"]]["balance"] += p["cashback_amount"] #CH
            p["received"] = True                     #SK

pay / 建立 scheduled event（完整 example）

def pay(self, timestamp, account_id, amount):        #SK
    self._process_cashbacks(timestamp)               #SK
    if account_id not in self.accounts: return None  #SK
    acc = self.accounts[account_id]                  #SK
    if acc["balance"] < amount: return None           #SK
    acc["balance"] -= amount                         #CH
    acc["outgoing"] += amount                        #CH  # L2 加
    acc["history"].append((timestamp, acc["balance"])) #CH # L4 加
    cashback = amount * 2 // 100                     #CH  # 計算 scheduled amount
    cashback_time = timestamp + 86400000             #CH  # 計算 deadline
    self.payment_counter += 1                        #SK
    pid = f"payment{self.payment_counter}"           #SK
    self.payment[pid] = {"account_id": account_id, "cashback_amount": cashback, "cashback_time": cashback_time, "received": False} #CH
    return pid                                       #SK

def get_payment_status(self, timestamp, account_id, payment_id): #SK

歷史查詢（完整 example）

def get_balance(self, timestamp, account_id, time_at): #SK
    self._process_cashbacks(timestamp)               #SK
    if account_id not in self.accounts: return None  #SK
    acc = self.accounts[account_id]                  #SK
    for ts, val in reversed(acc["history"]):          #SK  # 由尾搵返去
        if ts <= time_at:                            #SK

merge / 合併兩個 entity（完整 example）

def merge_accounts(self, timestamp, account_id_1, account_id_2): #SK
    self._process_cashbacks(timestamp)               #SK
    if account_id_1 not in self.accounts or account_id_2 not in self.accounts: return False #SK
    if account_id_1 == account_id_2: return False    #SK
    acc_1 = self.accounts[account_id_1]              #SK
    acc_2 = self.accounts[account_id_2]              #SK
    acc_1["balance"] += acc_2["balance"]              #CH  # 合併 data
    acc_1["outgoing"] += acc_2["outgoing"]            #CH
    acc_1["history"].append((timestamp, acc_1["balance"])) #CH
    self.merged_accounts[account_id_2] = account_id_1 #SK
    del self.accounts[account_id_2]                  #SK
    return True                                      #SK

L5 — Concurrent Batch Processing

三步：1. 定義 execute_op 2. gather 同時跑 3. return list(results)

async def process_batch(self, timestamp, operations): #SK
    self._process_cashbacks(timestamp)               #SK

    async def execute_op(op):                        #SK
        aid = op.get("account_id") or op.get("source_id") #SK
        lock = self.account_locks[aid]               #SK
        async with lock:                             #SK
            # -------- 以下因應題目改 --------
            if op["type"] == "deposit":              #CH
                return self.deposit(timestamp, op["account_id"], op["amount"]) #CH
            elif op["type"] == "transfer":           #CH
                return self.transfer(timestamp, op["source_id"], op["target_id"], op["amount"]) #CH
            elif op["type"] == "pay":                #CH
                return self.pay(timestamp, op["account_id"], op["amount"]) #CH
            # -------- 以上因應題目改 --------

    results = await asyncio.gather(*[execute_op(op) for op in operations]) #SK
    return list(results)                             #SK

L6 — Rate Limited External Call

同 L5 一樣，加 Semaphore。Lock 包住改 data，Sem 包住外部 call。唔好 return 太早。

async def process_external_transfers(self, timestamp, transfers, max_concurrent): #SK
    self._process_cashbacks(timestamp)               #SK
    sem = asyncio.Semaphore(max_concurrent)           #SK

    async def do_transfer(t):                        #SK
        aid = t["account_id"]                        #CH
        lock = self.account_locks[aid]               #SK
        async with lock:                             #SK  # 鎖住 account
            # -------- 以下因應題目改（check + 扣錢） --------
            if aid not in self.accounts:             #CH
                return False                         #CH
            if self.accounts[aid]["balance"] < t["amount"]: #CH
                return False                         #CH
            self.accounts[aid]["balance"] -= t["amount"] #CH
            self.accounts[aid]["outgoing"] += t["amount"] #CH
            self.accounts[aid]["history"].append((timestamp, self.accounts[aid]["balance"])) #CH
            # -------- 以上因應題目改 --------
        async with sem:                              #SK  # 限制同時數量
            await asyncio.sleep(0.01)                #SK  # 模擬外部 call
        return True                                  #SK

    results = await asyncio.gather(*[do_transfer(t) for t in transfers]) #SK
    return list(results)                             #SK
L5 vs L6 分別：
L5：gather + Lock（全部同時跑，鎖住 account）
L6：gather + Lock + Semaphore（限制同時幾多個 + 模擬外部 call）
Lock 同 Sem 分開用。Lock 包住改 data，Sem 包住外部 call。

Gotchas（做過嘅錯誤）

❌ _init_ 一個底線 → ✅ __init__ 兩個底線
❌ def method(account_id): 漏 self → ✅ def method(self, account_id):
❌ d.balance dot → ✅ d["balance"] bracket
❌ accounts.append(acc) → ✅ accounts[key] = acc dict 用 key
❌ balance <= amount return None → ✅ balance < amount 等於夠錢要畀轉
❌ for x in dict: x.key → ✅ x 已經係 key string
❌ payment = d[pid]: 多咗冒號 → ✅ 唔係 function 唔使冒號
❌ return IN_PROGRESS → ✅ return "IN_PROGRESS" 要引號
❌ acc.amount = amount 覆蓋 → ✅ acc["balance"] += amount 累加
❌ 邊 iterate 邊 delete dict → ✅ 先收集 keys 再 delete
❌ copy_ref = original 同一個 object → ✅ copy.deepcopy(original)
❌ async with lock: return ... 之後行唔到 → ✅ 唔好 return 太早
