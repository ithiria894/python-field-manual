def transfer(self, timestamp, source_id, target_id, amount):
    self._process_cashbacks(timestamp)     # L3 加

    if source_id not in self.accounts or target_id not in self.accounts:
        return None
    if source_id == target_id:
        return None

    s_acc = self.accounts[source_id]
    t_acc = self.accounts[target_id]
    if s_acc["balance"] < amount:
        return None

    s_acc["balance"] -= amount
    s_acc["outgoing"] += amount            # L2 加
    s_acc["history"].append((timestamp, s_acc["balance"])) # L4 加

    t_acc["balance"] += amount
    t_acc["history"].append((timestamp, t_acc["balance"])) # L4 加

    return s_acc["balance"]

def top_spenders(self, timestamp, n):
    self._process_cashbacks(timestamp)

    sorted_items = dict(sorted(
        self.accounts.items(),
        key=lambda x: (-x[1]["outgoing"], x[0])
    ))

    result = []
    for account_id, acc in sorted_items.items():
        result.append(f"{account_id}({acc['outgoing']})")

    return result[:n]

L3 — Timestamp / TTL / Lazy Processing

pay 扣錢 + schedule cashback（24h 後自動入賬）。每個 method 開頭 call _process_cashbacks 處理到期嘅。get_payment_status return "IN_PROGRESS" 或 "CASHBACK_RECEIVED"。

Timestamp = 「而家幾點」，數字，大 = 遲
TTL = 幾耐之後過期，expiry = timestamp + ttl
Lazy processing = 唔會自動到期，要等有人 call method 先 check

Lazy Processing Helper（完整 example）

所有題都用。唔會自動到期，每次有人 call method 帶 timestamp 入嚟先 check。

def _process_cashbacks(self, timestamp):
    for pid, p in self.payment.items():
        if not p["received"] and timestamp >= p["cashback_time"]:
            self.accounts[p["account_id"]]["balance"] += p["cashback_amount"]
            p["received"] = True

pay / 建立 scheduled event（完整 example）

def pay(self, timestamp, account_id, amount):
    self._process_cashbacks(timestamp)

    if account_id not in self.accounts: return None
    acc = self.accounts[account_id]
    if acc["balance"] < amount: return None

    acc["balance"] -= amount
    acc["outgoing"] += amount
    acc["history"].append((timestamp, acc["balance"]))

    cashback = amount * 2 // 100                   # ← 計算 scheduled amount
    cashback_time = timestamp + 86400000             # ← 計算 deadline

    self.payment_counter += 1
    pid = f"payment{self.payment_counter}"
    self.payment[pid] = {"account_id": account_id, "cashback_amount": cashback, "cashback_time": cashback_time, "received": False}

    return pid

get status（完整 example）

def get_payment_status(self, timestamp, account_id, payment_id):
    self._process_cashbacks(timestamp)

    if account_id not in self.accounts: return None
    if payment_id not in self.payment: return None
    if self.payment[payment_id]["account_id"] != account_id: return None
    if self.payment[payment_id]["received"]:
        return "CASHBACK_RECEIVED"
    return "IN_PROGRESS"

L4 — Backup / Merge / Historical Query

get_balance(time_at) 查歷史某個時間嘅 balance（reversed loop history）。merge 合併兩個 account（balance + outgoing 加埋，del 第二個）。

歷史查詢（完整 example）

def get_balance(self, timestamp, account_id, time_at):
    self._process_cashbacks(timestamp)

    if account_id not in self.accounts: return None

    acc = self.accounts[account_id]
    for ts, val in reversed(acc["history"]):      # ← 由尾搵返去
        if ts <= time_at:
            return val

merge / 合併兩個 entity（完整 example）

def merge_accounts(self, timestamp, account_id_1, account_id_2):
    self._process_cashbacks(timestamp)

    if account_id_1 not in self.accounts or account_id_2 not in self.accounts: return False
    if account_id_1 == account_id_2: return False

    acc_1 = self.accounts[account_id_1]
    acc_2 = self.accounts[account_id_2]

    acc_1["balance"] += acc_2["balance"]              # ← 合併 data
    acc_1["outgoing"] += acc_2["outgoing"]
    acc_1["history"].append((timestamp, acc_1["balance"]))

    self.merged_accounts[account_id_2] = account_id_1
    del self.accounts[account_id_2]

    return True

L5 — Concurrent Batch Processing

同時處理多個 operations。Skeleton：async def + execute_op + gather + lock。入面嘅 if/elif 因應題目改。抄 template 改幾行就得。

完整 example

三步：1. 定義 execute_op 2. gather 同時跑 3. return list(results)

async def process_batch(self, timestamp, operations):
    self._process_cashbacks(timestamp)

    async def execute_op(op):
        aid = op.get("account_id") or op.get("source_id")
        lock = self.account_locks[aid]
        async with lock:
            # -------- 以下因應題目改 --------
            if op["type"] == "deposit":
                return self.deposit(timestamp, op["account_id"], op["amount"])
            elif op["type"] == "transfer":
                return self.transfer(timestamp, op["source_id"], op["target_id"], op["amount"])
            elif op["type"] == "pay":
                return self.pay(timestamp, op["account_id"], op["amount"])
            # -------- 以上因應題目改 --------

    results = await asyncio.gather(*[execute_op(op) for op in operations])
    return list(results)

L6 — Rate Limited External Call

同 L5 一樣但加 Semaphore 限制同時幾多個。Lock 包住改 data，Sem 包住外部 call（sleep）。Failed 嘅 return False 唔 sleep。抄 template 改 check + 扣錢嗰段就得。

完整 example

同 L5 一樣，加 Semaphore。Lock 包住改 data，Sem 包住外部 call。唔好 return 太早。

async def process_external_transfers(self, timestamp, transfers, max_concurrent):
    self._process_cashbacks(timestamp)
    sem = asyncio.Semaphore(max_concurrent)

    async def do_transfer(t):
        aid = t["account_id"]
        lock = self.account_locks[aid]

        async with lock:                               # ← 鎖住 account
            # -------- 以下因應題目改（check + 扣錢） --------
            if aid not in self.accounts:
                return False
            if self.accounts[aid]["balance"] < t["amount"]:
                return False

            self.accounts[aid]["balance"] -= t["amount"]
            self.accounts[aid]["outgoing"] += t["amount"]
            self.accounts[aid]["history"].append((timestamp, self.accounts[aid]["balance"]))
            # -------- 以上因應題目改 --------

        async with sem:                                # ← 限制同時數量
            await asyncio.sleep(0.01)                  # ← 模擬外部 call

        return True

    results = await asyncio.gather(*[do_transfer(t) for t in transfers])
    return list(results)
L5 vs L6 分別：
L5：gather + Lock（全部同時跑，鎖住 account）
L6：gather + Lock + Semaphore（限制同時幾多個 + 模擬外部 call）

Lock 同 Sem 分開用。Lock 包住改 data，Sem 包住外部 call。

Gotchas（做過嘅錯誤）

❌ _init_ 一個底線 → ✅ __init__ 兩個底線
❌ def method(account_id): 漏 self → ✅ def method(self, account_id):
❌ d.balance dot → ✅ d["balance"] bracket
❌ accounts.append(acc) → ✅ accounts[key] = acc dict 用 key
❌ balance <= amount return None → ✅ balance < amount 等於夠錢要畀轉
❌ for x in dict: x.key → ✅ x 已經係 key string
❌ payment = d[pid]: 多咗冒號 → ✅ 唔係 function 唔使冒號
❌ return IN_PROGRESS → ✅ return "IN_PROGRESS" 要引號
❌ acc.amount = amount 覆蓋 → ✅ acc["balance"] += amount 累加
❌ 邊 iterate 邊 delete dict → ✅ 先收集 keys 再 delete
❌ copy_ref = original 同一個 object → ✅ copy.deepcopy(original)
❌ async with lock: return ... 之後嘅 code 行唔到 → ✅ 唔好 return 太早，lock 同 sem 分開

def book_room(self, timestamp, room_id, guest_name, nights):
    self._process_late_fee(timestamp)

    if room_id not in self.rooms: return None
    room = self.rooms[room_id]
    if room["guest_name"] != "": return None

    room["guest_name"] = guest_name
    room["nights"] = nights
    room["total_revenue"] += room["price_per_night"] * nights
    room["check_in_time"] = timestamp

    self.booking_counter += 1
    booking_id_str = f"booking{self.booking_counter}"
    room["booking_id"] = booking_id_str

    return booking_id_str
def checkout(self, timestamp, room_id):
    self._process_late_fee(timestamp)

    if room_id not in self.rooms: return None
    room = self.rooms[room_id]
    if room["guest_name"] == "": return None

    guest_name = room["guest_name"]
    room["guest_name"] = ""
    room["history"].append((guest_name, room["price_per_night"] * room["nights"])) # L4 加
    room["nights"] = 0

    return guest_name

def late_checkout(self, timestamp, room_id, extra_hours):
    self._process_late_fee(timestamp)

    if room_id not in self.rooms: return None
    room = self.rooms[room_id]
    if room["guest_name"] == "": return None

    fee = extra_hours * 50
    room["late_fee"] = fee                     # 擺喺 room 入面
    room["late_fee_time"] = timestamp + 3600000

    self.checkout(timestamp, room_id)

    return fee
def _process_late_fee(self, timestamp):
    for room_id, room in self.rooms.items():
        if room["late_fee"] > 0 and timestamp >= room["late_fee_time"]:
            room["total_revenue"] += room["late_fee"]
            room["late_fee"] = 0
def get_pending_fees(self, timestamp):
    self._process_late_fee(timestamp)
    all_fee = 0
    for room_id, room in self.rooms.items():
        if room["late_fee"] != 0:
            all_fee += room["late_fee"]
    return all_fee

def get_booking_history(self, timestamp, room_id):
    self._process_late_fee(timestamp)

    if room_id not in self.rooms: return None

    room = self.rooms[room_id]
    list = []

def upgrade_room(self, timestamp, from_room_id, to_room_id):
    self._process_late_fee(timestamp)

    if from_room_id not in self.rooms or to_room_id not in self.rooms: return False
    if from_room_id == to_room_id: return False

    from_room = self.rooms[from_room_id]
    to_room = self.rooms[to_room_id]

    if from_room["guest_name"] == "": return False   # from 要有人
    if to_room["guest_name"] != "": return False    # to 要冇人

    # 直接搬，唔好 call checkout
    to_room["guest_name"] = from_room["guest_name"]
    to_room["booking_id"] = from_room["booking_id"]
    to_room["nights"] = from_room["nights"]
    from_room["guest_name"] = ""

    return True

    async def execute_op(op):

        lock = self.rooms_locks[rid]
        async with lock:
            if op["type"] == "book":
                return self.book_room(timestamp, op["room_id"], op["guest_name"], op["nights"])
            elif op["type"] == "checkout":
                return self.checkout(timestamp, op["room_id"])
            elif op["type"] == "upgrade":
                return self.upgrade_room(timestamp, op["from_room_id"], op["to_room_id"])

    results = await asyncio.gather(*[execute_op(op) for op in operations])
    return list(results)

L6 — Rate Limited Notifications（Hotel）

同時 send notifications。Failed（room 唔存在或 guest 唔啱）唔 sleep。

async def send_notifications(self, timestamp, notifications, max_concurrent):
    self._process_late_fee(timestamp)
    sem = asyncio.Semaphore(max_concurrent)

    async def do_notification(t):
        rid = t["room_id"]
        lock = self.rooms_locks[rid]

        async with lock:
            if rid not in self.rooms:
                return False
            if self.rooms[rid]["guest_name"] != t["guest_name"]:
                return False
            if self.rooms[rid]["guest_name"] == "":
                return False

        async with sem:                       # 過咗 check 先用 sem
            await asyncio.sleep(0.01)

        return True

    results = await asyncio.gather(*[do_notification(t) for t in notifications])
    return list(results)

def set(self, key, field, value):
    if key not in self.data:
        self.data[key] = {}           # 開新行
    self.data[key][field] = {
        "value": value,
        "expiry": None              # L3 會用

def get(self, key, field):
    if key not in self.data or field not in self.data[key]:
        return ""                  # 唔係 None！

    if not self.data[key]:          # 行冇欄了
        del self.data[key]        # 刪行
    return True

def scan(self, key):

L3 — TTL (set_at / set_at_with_ttl / get_at / delete_at / scan_at)

唔用 lazy helper。用 inline _is_alive check。過期嘅唔真刪，只係跳過。

# helper：呢個 field 而家仲活唔活？
def _is_alive(self, field_data, timestamp=None):
    if field_data["expiry"] is None:    # 冇 TTL → 永遠活
        return True
    if timestamp is None:              # L1/L2 冇 ts → 當活
        return True
    return timestamp < field_data["expiry"]  # < 唔係 <=

def set_at(self, key, field, value, timestamp):
    if key not in self.data:
        self.data[key] = {}
    self.data[key][field] = {"value": value, "expiry": None}

def set_at_with_ttl(self, key, field, value, timestamp, ttl):
    if key not in self.data:
        self.data[key] = {}
    self.data[key][field] = {"value": value, "expiry": timestamp + ttl}

def get_at(self, key, field, timestamp):
    if key not in self.data or field not in self.data[key]:
        return ""
    fd = self.data[key][field]
    if not self._is_alive(fd, timestamp):
        return ""                       # 過期 → 當冇
    return fd["value"]

def delete_at(self, key, field, timestamp):
    if key not in self.data or field not in self.data[key]:
        return False
    if not self._is_alive(self.data[key][field], timestamp):
        return False                    # 已死 = 當唔存在
    del self.data[key][field]
    if not self.data[key]: del self.data[key]
    return True

def scan_at(self, key, timestamp):
    if key not in self.data:
        return ""
    fields = sorted(
        f for f in self.data[key]
        if self._is_alive(self.data[key][f], timestamp)
    )
    if not fields: return ""
    parts = [f"{f}({self.data[key][f]['value']})" for f in fields]
    return ", ".join(parts)

def scan_by_prefix_at(self, key, prefix, timestamp):
    if key not in self.data:
        return ""
    fields = sorted(
        f for f in self.data[key]
        if f.startswith(prefix) and self._is_alive(self.data[key][f], timestamp)
    )
    if not fields: return ""
    parts = [f"{f}({self.data[key][f]['value']})" for f in fields]
    return ", ".join(parts)

L4 — Backup & Restore

backup 存 remaining_ttl（唔係 expiry！），restore 用 remaining_ttl 重新計 expiry

def backup(self, timestamp):
    snapshot = {}
    count = 0
    for key, fields in self.data.items():
        alive_fields = {}
        for field, fd in fields.items():
            if self._is_alive(fd, timestamp):
                remaining = None
                if fd["expiry"] is not None:
                    remaining = fd["expiry"] - timestamp  # 仲剩幾耐
                alive_fields[field] = {
                    "value": fd["value"],
                    "remaining_ttl": remaining
                }
        if alive_fields:
            snapshot[key] = alive_fields
            count += 1
    self.backups.append((timestamp, snapshot))
    return count

def restore(self, timestamp, backup_timestamp):
    best = None
    for ts, snap in self.backups:
        if ts <= backup_timestamp:         # at or before
            if best is None or ts > best[0]:
                best = (ts, snap)
    if best is None:
        return ""

    _, snapshot = best
    self.data = {}                          # 清空
    count = 0
    for key, fields in snapshot.items():
        self.data[key] = {}
        for field, fd in fields.items():
            expiry = None
            if fd["remaining_ttl"] is not None:
                expiry = timestamp + fd["remaining_ttl"]  # 重算
            self.data[key][field] = {
                "value": fd["value"],
                "expiry": expiry
            }
        if self.data[key]: count += 1
    return str(count)                      # return string！

L5 — Batch (batch_operations)

同 Bank L5 一模一樣嘅 pattern。lock per key。set return None。

async def batch_operations(self, operations):
    async def execute_op(op):
        key = op.get("key", "")
        lock = self.key_locks[key]
        async with lock:
            if op["type"] == "set":
                self.set(op["key"], op["field"], op["value"])
                return None                 # set 冇 return
            elif op["type"] == "get":
                return self.get(op["key"], op["field"])
            elif op["type"] == "delete":
                return self.delete(op["key"], op["field"])
            elif op["type"] == "scan":
                return self.scan(op["key"])
        return None

    results = await asyncio.gather(*[execute_op(op) for op in operations])
    return list(results)

pattern（同 Bank 一樣）：
# 1. lock per KEY（唔係 per field）
# 2. async with lock: 入面 call sync method
# 3. asyncio.gather(*[...]) 同時跑
# 4. set return None

L6 — Rate-Limited Scan (batch_scan)

All-sleep — 全部 key 都 sleep，冇 fail-fast。同 Bank L6 唔同！

async def batch_scan(self, keys, max_concurrent):
    sem = asyncio.Semaphore(max_concurrent)

    async def scan_key(key):
        async with sem:                  # 全部都攞 sem
            await asyncio.sleep(0.01)    # 全部都 sleep
            return (key, self.scan(key))

    results = await asyncio.gather(*[scan_key(k) for k in keys])
    return dict(results)             # return dict 唔係 list
All-sleep vs Fail-fast：
# Bank L6 (fail-fast):
#   check 喺 sem 之前
#   失敗 → return False，唔 sleep
#
# InMemDB L6 (all-sleep):
#   冇 check，全部 sleep
#   key 唔存在？scan return "" → 照 sleep
例子：
keys = ["u1", "u2", "nokey"]
max_concurrent = 2

# 第一輪（同時 2 個）: u1 + u2 → sleep
# 第二輪: nokey → sleep
→ {"u1":"name(alice)","u2":"name(bob)","nokey":""}

# return dict，唔係 list
點分 fail-fast vs all-sleep：
# spec 講 "skip missing without I/O" → fail-fast
# spec 講 "scan all keys" → all-sleep
# InMemDB: 冇 fail path → all-sleep
